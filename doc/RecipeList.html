def listbuilder
recipes = result.collect {|recipe| recipe.name}
elements = result.collect {|recipe| recipe.ingredients}
        for num in 0..recipes.size-1
                @recipehash[recipes[num]]=elements[num]
        end
@recipehash.each {|k,v| v<<0}
end
